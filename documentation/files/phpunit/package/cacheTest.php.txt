require_once 'unitInit.php';

class cacheTest extends PHPUnit_Framework_TestCase
{
	private $cache;

	public function setUp()
	{
		$this->cache	=	\package\autoload::get('cache', '\package\\');
	}

	public function testSetCacheDir()
	{
		if($this->cache instanceof \package\cache)
		{
			$created	=	$this->cache->setCacheDir('createCacheDir');

			$this->assertEquals(true, $created);

			if($created === true)
			{
				rmdir('createCacheDir');
			}
		}
	}


	public function testSetAndGetAndDeleteElement()
	{
		if($this->cache instanceof \package\cache)
		{
			$created	=	$this->cache->setCacheDir('createCacheDir/');

			$this->assertEquals(true, $created);

			if($created === true)
			{
				$setElement	=	$this->cache->setElement('test', 'PHPUnitTest', 1000);

				$this->assertEquals(true, $setElement);

				if($setElement === true)
				{
					$getElement	=	$this->cache->getElement('test');

					$this->assertEquals('PHPUnitTest', $getElement);

					$deleteElement	=	$this->cache->deleteElement('test');

					$this->assertEquals(true, $deleteElement);
				}

				rmdir('createCacheDir');
			}
		}
	}
}
