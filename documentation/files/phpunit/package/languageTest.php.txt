require_once 'unitInit.php';

class languageTest extends PHPUnit_Framework_TestCase
{
	public function setUp()
	{
		\package\autoload::get('language', '\package\\', true);
	}


	public function testSetAndGetLanguage()
	{
		$this->assertEmpty(\package\language::getLanguage());

		\package\language::setLanguage('en_US');

		$this->assertEquals('en_US', \package\language::getLanguage());
	}


	public function testSetAndGetDefaultLanguage()
	{
		$this->assertEquals('de_DE', \package\language::getDefaultLanguage());

		\package\language::setDefaultLanguage('en_US');

		$this->assertEquals('en_US', \package\language::getDefaultLanguage());
	}


	public function testLoadLang()
	{
		\package\language::setLanguagePath('test');
		$this->assertTrue(\package\language::loadLang(true));
	}


	public function testTranslate()
	{
		\package\language::setLanguagePath('test');
		$this->assertTrue(\package\language::loadLang(true));

		$this->assertEquals('Robbyn', \package\language::translate('Robbyn'));
	}
}
