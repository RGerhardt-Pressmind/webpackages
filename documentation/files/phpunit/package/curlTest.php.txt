require_once 'unitInit.php';

class curlTest extends PHPUnit_Framework_TestCase
{
	private $curl;

	public function setUp()
	{
		$this->curl	=	\package\autoload::get('curl', '\package\\');
	}


	public function testExistsCurl()
	{
		if($this->curl instanceof \package\curl)
		{
			$this->assertTrue($this->curl->existCurl());
		}
	}


	public function testGetData()
	{
		if($this->curl instanceof \package\curl)
		{
			$data	=	$this->curl->getData('http://www.worldwideboard.de');

			$this->assertNotEmpty($data);
		}
	}


	public function testGetState()
	{
		if($this->curl instanceof \package\curl)
		{
			$data	=	$this->curl->getState('http://www.worldwideboard.de');

			$this->assertEquals(200, $data);
		}
	}


	public function testGetCityCoordinates()
	{
		if($this->curl instanceof \package\curl)
		{
			$data	=	$this->curl->getCityCoordinates('Berlin');

			$this->assertEquals(52.5200066, $data->lat);
			$this->assertEquals(13.404954, $data->lng);
		}
	}


	public function testGetCityName()
	{
		if($this->curl instanceof \package\curl)
		{
			$data	=	$this->curl->getCityNameByIp();

			$this->assertNotEmpty($data);
		}
	}
}
